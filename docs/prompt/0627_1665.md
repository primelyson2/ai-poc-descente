```sql
-- change_reason: 상관 서브쿼리를 CTE와 조인으로 변경하여 중복 집계 실행 방지
-- change_summary: 연도별 카테고리 평균 amount_sum을 별도 CTE로 계산 후 z와 조인하여 필터링
-- change_location: 외부 WHERE 절의 상관 서브쿼리 부분
WITH yearly_avg AS (
  SELECT calendar_year, AVG(amount_sum) AS avg_amount
  FROM (
    SELECT p.prod_category, t.calendar_year, SUM(s.amount_sold) AS amount_sum
    FROM DEVDO.SALES s
    JOIN DEVDO.PRODUCTS p ON p.prod_id = s.prod_id
    JOIN DEVDO.TIMES t ON t.time_id = s.time_id
    WHERE t.calendar_year IN (1998, 1999, 2000, 2001)
    GROUP BY p.prod_category, t.calendar_year
  ) y
  GROUP BY calendar_year
),
 z AS (
  SELECT co.country_region,
    p.prod_category,
    ch.channel_desc,
    t.calendar_year,
    COUNT(*) sales_cnt,
    SUM(s.amount_sold) amount_sum,
    SUM(s.quantity_sold) qty_sum,
    COUNT(DISTINCT c.cust_id) customer_cnt,
    AVG(c.cust_year_of_birth) avg_customer_birth_year
  FROM DEVDO.SALES s
  JOIN DEVDO.PRODUCTS p ON p.prod_id = s.prod_id
  JOIN DEVDO.CUSTOMERS c ON c.cust_id = s.cust_id
  JOIN DEVDO.COUNTRIES co ON co.country_id = c.country_id
  JOIN DEVDO.TIMES t ON t.time_id = s.time_id
  LEFT JOIN DEVDO.CHANNELS ch ON ch.channel_id = s.channel_id
  LEFT JOIN DEVDO.PROMOTIONS pr ON pr.promo_id = s.promo_id
  WHERE t.calendar_year IN (1998, 1999, 2000, 2001)
    AND SUBSTR(UPPER(NVL(co.country_region, 'UNKNOWN')), 1, 3) IN ('AME', 'ASI', 'EUR')
    AND UPPER(NVL(p.prod_category, 'X')) LIKE UPPER('%a%')
    AND NVL(pr.promo_cost, 0) >= 0
  GROUP BY co.country_region, p.prod_category, ch.channel_desc, t.calendar_year
)
SELECT z.country_region,
  z.prod_category,
  z.channel_desc,
  z.calendar_year,
  z.sales_cnt,
  z.amount_sum,
  z.qty_sum,
  z.customer_cnt,
  z.avg_customer_birth_year,
  DENSE_RANK() OVER (PARTITION BY z.country_region, z.calendar_year ORDER BY z.amount_sum DESC NULLS LAST) region_rank
FROM z
JOIN yearly_avg ya ON ya.calendar_year = z.calendar_year
WHERE z.amount_sum > ya.avg_amount
ORDER BY z.calendar_year, z.country_region, region_rank
```
